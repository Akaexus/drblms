Module: ipCalc

Moduł do wygodnego przeprowadzania adresacji IP w wersji 4.

Version:
  • 0.2.0
Source:
  • libs/ipCalc.js, line 4
Example
const ipCalc = require('./libs/ipCalc.js');
var ip = new ipCalc('10.0.21.37', '255.0.0.0');
console.log(ip.info());

// { address: [ 10, 0, 0, 0 ],
//   cidr: 8,
//   mask: [ 255, 0, 0, 0 ],
//   networkAddress: [ 10, 0, 0, 0 ],
//   broadcastAddress: [ 10, 255, 255, 255 ],
//   numberOfHosts: 16777214 }
